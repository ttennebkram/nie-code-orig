134     /**
135      * Updates to the tree that we need to do on any change
136      */
137     protected void updateTree()
138     {
139         // expand the tree
140         int row = 0;
141         while (row < tree.getRowCount())
142         {
143             tree.expandRow(row++);
144         }
145 
146         ctm.fireTreeStructureChanged(this);
147     }

149     /**
150      * Add a Choice to our set of panels
151      */
152     protected void addChoice(String key, Choice model)
153     {
154         super.addChoice(key, model);
155 
156         // Sort the tree out
157         String path = Config.getPath(model.getFullPath());
158         FormPane card = (FormPane) decks.get(path);
159         if (card != null && card.getParent() == null)
160         {
161             JScrollPane scroll = new JScrollPane(card);
162             scroll.setBorder(BorderFactory.createEmptyBorder());
163             deck.add(path, scroll);
164         }
165     }
166 
167     /**
168      * Add a Choice to our set of panels
169      */
170     protected void removeChoice(String key, Choice model)
171     {
172         super.removeChoice(key, model);
173 
174         // Sort the tree out
175         String path = Config.getPath(model.getFullPath());
176         FormPane card = (FormPane) decks.get(path);
177         if (card != null && card.isEmpty())
178         {
179             deck.remove(card.getParent());
180         }
181     }

183     /**
184      * Used to update the configuration panel whenever someone
185      * selects a different item form the tree on the LHS of the
186      * configuation dialog.
187      */
188     public void selectCard()
189     {
190         Object obj = tree.getLastSelectedPathComponent();
191         if (obj == null)
192         {
193             return;
194         }
195 
196         title.setText(obj + " " + Msg.PROPERTIES); //$NON-NLS-1$
197 
198         // Get the name of the current deck
199         Object[] list = tree.getSelectionPath().getPath();
200         StringBuffer path = new StringBuffer();
201 
202         for (int i = 1; i < list.length; i++)
203         {
204             if (i > 1)
205             {
206                 path.append("."); //$NON-NLS-1$
207             }
208 
209             path.append(list[i].toString());
210         }
211 
212         String key = path.toString();
213         if (decks.containsKey(key))
214         {
215             layout.show(deck, key);
216         }
217         else
218         {
219             layout.show(deck, BLANK);
220         }
221 
222         deck.repaint();
223     }

275         /**
276          * Get a Vector of the children rooted at path
277          */
278         protected List getChildren(String path)
279         {
280             List retcode = new ArrayList();
281 
282             Iterator it = config.getPaths();
283             while (it.hasNext())
284             {
285                 String temp = (String) it.next();
286 
287                 if (temp.startsWith(path) && !temp.equals(path))
288                 {
289                     // Chop off the similar start
290                     temp = temp.substring(path.length());
291                     if (temp.startsWith(".")) //$NON-NLS-1$
292                     {
293                         temp = temp.substring(1);
294                     }
295 
296                     // Chop off all after the first dot
297                     int dot_pos = temp.indexOf("."); //$NON-NLS-1$
298                     if (dot_pos != -1)
299                     {
300                         temp = temp.substring(0, dot_pos);
301                     }
302 
303                     // Add it to the list if needed
304                     if (temp.length() > 0 && !retcode.contains(temp))
305                     {
306                         retcode.add(temp);
307                     }
308                 }
309             }
310 
311             return retcode;
312         }

375         /**
376          * Notify all listeners that have registered interest for
377          * notification on this event type.  The event instance
378          * is lazily created using the parameters passed into
379          * the fire method.
380          * @see EventListenerList
381          */
382         protected void fireTreeStructureChanged(Object source)
383         {
384             fireTreeStructureChanged(source, new Object[] { root });
385         }
386 
387         /**
388          * Notify all listeners that have registered interest for
389          * notification on this event type.  The event instance
390          * is lazily created using the parameters passed into
391          * the fire method.
392          * @see EventListenerList
393          */
394         protected void fireTreeStructureChanged(Object source, Object[] path)
395         {
396             // Guaranteed to return a non-null array
397             Object[] array = listeners.getListenerList();
398             TreeModelEvent ev = null;
399 
400             // Process the listeners last to first, notifying
401             // those that are interested in this event
402             for (int i = array.length - 2; i >= 0; i -= 2)
403             {
404                 if (array[i] == TreeModelListener.class)
405                 {
406                     // Lazily create the event:
407                     if (ev == null)
408                     {
409                         ev = new TreeModelEvent(source, path);
410                     }
411 
412                     ((TreeModelListener) array[i + 1]).treeStructureChanged(ev);
413                 }
414             }
415         }

428     /**
429      * Simple Tree Node
430      */
431     protected static class Node
432     {
433         /**
434          * Create a node with a name and path
435          */
436         protected Node(String path, String name)
437         {
438             this.path = path;
439             this.name = name;
440         }
441 
442         /* (non-Javadoc)
443          * @see java.lang.Object#toString()
444          */
445         public String toString()
446         {
447             return name;
448         }
449 
450         /**
451          * The path to us
452          */
453         public String getFullName()
454         {
455             if (path.length() == 0 || name.length() == 0)
456             {
457                 return path + name;
458             }
459             return path + "." + name; //$NON-NLS-1$
460         }
461 
462         /**
463          * The displayed string
464          */
465         private String name;
466 
467         /**
468          * The path to us
469          */
470         private String path;
